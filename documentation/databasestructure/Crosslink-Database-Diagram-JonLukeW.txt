// cross-link
Table users {
  id int [primary key]
  username varchar(50) [not null, unique]
  password varchar(512) [not null]
}

Table bookmarks {
  id int [primary key]
  title varchar(255)
  url varchar(512)
  description text
  thumbnail text // URI or Base64 encoded?
  user_id int [ref: > users.id]
}

Table tags {
  id int [primary key]
  name varchar(50)
}

Table bookmark_tag {
  bookmark_id int [primary key, ref: > bookmarks.id]
  tag_id int [primary key, ref: > tags.id]
}

Table roles {
  id int [primary key]
  name varchar(50) [not null, unique]
  description varchar(255)
}

Table role_user {
  user_id int [primary key, ref: > users.id]
  role_id int [primary key, ref: > roles.id]
}

Table profiles {
  user_id int [primary key, ref: - users.id]
  photo text // URI or Base64 encoded?
  email varchar(320)
  first_name varchar(128)
  last_name varchar(128)
}

Table social_media_links {
  id int [primary key]
  name varchar(50)
  url varchar(512)
  profile_id int [ref: > profiles.user_id]
}
